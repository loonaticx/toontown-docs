[docs]def isThought(message):
    """
    message is a string.

    Return 1 if the given string contains the thought prefix,
    Return 0 otherwise
    """
    if (len(message) == 0):
        # empty string cannot be a thought
        return 0
    elif (message.find(ThoughtPrefix, 0,
                      len(ThoughtPrefix)) >= 0):
        return 1
    else:
        return 0



[docs]def removeThoughtPrefix(message):
    """
    message is a string.

    Return the string with the thought prefix removed
    """
    if (isThought(message)):
        return message[len(ThoughtPrefix):]
    else:
        return message
